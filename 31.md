# basic function for t.test of norm

``` r
norm_sim <- function(mu, n = 30, sd = 5) {
  sim_data=tibble(
    x <- rnorm(n,mean = mu, sd = sd))
  
  sim_data %>%
    t.test(mu = mu) %>%
    broom::tidy() %>%
    select(estimate,p.value)
}
```

``` r
sim_results_df = 
  expand_grid(
    mu=0:6,
    iter = 1:5000
  ) %>% 
  mutate(
    estimate_df = map(mu, norm_sim)
  ) %>% 
  unnest(estimate_df)
```

``` r
df1=sim_results_df%>%
  group_by(mu)%>%
```
